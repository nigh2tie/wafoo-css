// 和風グリーティング関数
function greet(name) {
  const greetings = ['こんにちは', 'おはよう', 'こんばんは'];
  const random = Math.floor(Math.random() * greetings.length);
  return `${greetings[random]}、${name}さん！`;
}

// 使用例
console.log(greet('太郎'));  // "こんにちは、太郎さん！"

通知システムの実装例

以下のコードで簡単に通知システムを実装できます：

function showToast(type) {
  const container = document.getElementById('toast-container');
  const toast = document.createElement('div');
  toast.className = `wf-toast${type !== 'info' ? ' wf-toast--' + type : ''}`;

  const messages = {
    info: { title: 'お知らせ', text: '新しい情報があります' },
    success: { title: '成功', text: 'データが正常に保存されました' },
    warning: { title: '注意', text: 'この操作は元に戻せません' },
    danger: { title: 'エラー', text: '処理中にエラーが発生しました' }
  };

  const msg = messages[type];
  toast.innerHTML = `
    <div class="wf-toast__content">
      <div class="wf-toast__title">${msg.title}</div>
      ${msg.text}
    </div>
    <button class="wf-toast__close" aria-label="閉じる">×</button>
  `;

  container.appendChild(toast);

  // 閉じるボタン
  toast.querySelector('.wf-toast__close').addEventListener('click', () => {
    toast.style.animation = 'wf-toast-out 0.2s ease forwards';
    setTimeout(() => toast.remove(), 200);
  });

  // 自動削除（5秒後）
  setTimeout(() => {
    if (toast.parentElement) {
      toast.style.animation = 'wf-toast-out 0.2s ease forwards';
      setTimeout(() => toast.remove(), 200);
    }
  }, 5000);
}
